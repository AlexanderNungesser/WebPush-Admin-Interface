Multilanguage support is allready here!

SWAC supports multi language support and it's allready configured for you.
You can add new entries to languages for a smooth translation by:

  1. Open /data/languages/app_de.js or /data/languages/app_en.js
  2. Write a new keywort for language entry
  3. Add the translation for your entry
  4. Use your new keyword within swag_lang-Attribute or inside the routes.json

var app_de = {
    howtos: 'Wie man...',
    howto_simple_page: 'eine einfache Seite erstellt',
    howto_sitemap: 'eine Sitemap erstellt',
    howto_language: 'eine Sprache einstellt',
    predefined_pages: 'Vorgefertigte Seiten',
    predefined_startpage: 'Startseite',
    predefined_imprint: 'Impressum',
    predefined_privacy: 'Datenschutzerklärung'
};
export default app_de;
